61def run_cpp_sort(algorithm: str) -> pd.DataFrame:
62 """
63 @brief Запускает сортировку в C++ и замеряет время выполнения.
64
65 @param algorithm Название алгоритма сортировки (insertion, heap, merge).
66 @return DataFrame с результатами замеров времени.
67 """
68 results = {"size": [], "time": [], "algo": []}
69 for size in tqdm(DATA_SIZES, desc=f"Sorting with {algorithm}..."):
70 process = subprocess.run(
71 ["./build/sort_ships", f"data/raw/ships_{size}.csv", algorithm],
72 capture_output=True,
73 text=True,
74 )
75
76 output = process.stdout.strip().split("\n")
77 time = float(output[1].split(":")[1].split()[0])
78 results["size"].append(size)
79 results["time"].append(time)
80 results["algo"].append(algorithm)
81
82 return pd.DataFrame(results)
